def train(epoch):
    """
    训练模型
    """
    plt.cla()
    epoch_loss = 0
    epoch_TPandFP = 0  # 预测为正样本的数量
    epoch_TPandFN = 0  # 实际为正样本的数量
    epoch_TP = 0  # 预测正确的数量

    model.train()

    # 确定数值，方便输出信息
    if config.is_load_model:
        epoch_num = config.model_load_epoch + epoch + 1
    else:
        epoch_num = epoch + 1
    print("-" * 129)

    #############################################################
    for batch, (x, label) in enumerate(data_set['train']):
        # 训练
        output, loss = propagate(x, label, "train")
        epoch_loss += loss * label.shape[0]

        # 统计数据
        pred = torch.gt(output, 0.5).float().cpu()  # 得出预测标签
        epoch_TPandFP += torch.sum(pred, dim=0)
        epoch_TPandFN += torch.sum(label, dim=0)
        epoch_TP += torch.sum(torch.eq(pred, label).mul(pred), dim=0)

        # 打印训练信息
        print("\r*TRAIN*",
              "[EPOCH]: {:04d}".format(epoch_num),
              '|' + ('▋' * int(batch / (len(data_set['train']) - 1) * 60)).ljust(60, '-') +
              '| {:>6.1%}'.format(batch / (len(data_set['train']) - 1)),
              end=' '
              )
    #############################################################

    # 学习率衰减
    if config.is_lr_decay and epoch_num % config.lr_decay_epoch_interval == 0:
        print("> Learning rate is decayed.")
        for param_group in optimizer.param_groups:
            param_group['lr'] *= config.lr_decay_coeff

    # 保存模型
    if config.is_save_model and (epoch + 1) % config.save_epoch_interval == 0:
        path = os.path.join(config.model_save_dir,
                            config.model_save_name.format(epoch_num))
        try:
            os.makedirs(config.model_save_dir)
        finally:
            torch.save(model.state_dict(), path)
        print("> CHECKPOINT saved to:")
        print("  " + path)
        print("-" * 50)

    # 记录每个周期的训练信息
    single_loss = epoch_loss / len(data_set['train'].dataset)
    precision = torch.div(epoch_TP, epoch_TPandFP)
    precision = torch.where(torch.isnan(precision), torch.full_like(precision, 0), precision).numpy().tolist()
    recall = torch.div(epoch_TP, epoch_TPandFN).numpy().tolist()
    F1_score = torch.div((2 * epoch_TP), (epoch_TPandFP + epoch_TPandFN)).numpy().tolist()
    avg_F1_score = torch.mean(torch.div((2 * epoch_TP), (epoch_TPandFP + epoch_TPandFN))).item()
    print("[F1-SCORE]: %0.6f [LOSS]: %0.6f" % (avg_F1_score, single_loss))
    writer.add_scalar('loss/train', single_loss, epoch_num)
    writer.add_scalars('precision/train', dict(zip(config.classes, precision)), epoch_num)
    writer.add_scalars('recall/train', dict(zip(config.classes, recall)), epoch_num)
    writer.add_scalar('F1-score/train_avg', avg_F1_score, epoch_num)
    writer.add_scalars('F1-score/train', dict(zip(config.classes, F1_score)), epoch_num)
    return single_loss, F1_score
